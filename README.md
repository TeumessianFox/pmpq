# Polynomial multiplication for post-quantum algorithms
Based on fork of https://github.com/libopencm3/libopencm3-template.git

# Build
 1. git clone git@github.com:TeumessianFox/pmpq.git
 2. cd pmpq
 3. git submodule update --init # (Only needed once)
 4. make -C libopencm3 # (Only needed once)
 5. make -C polymul
